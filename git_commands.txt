
git init [for initializing git repository]

git status [list all untracked files]

git add -A [adding all untracked files to staging area]

git commit -m "commit message" [ staged files to local repository]

git branch [list all branches]

git remote add name path [add a new remote repository]

ex:
    >git remote add origin https://github.com/luminartechnolabkochi/gitlearningmarch.git


git remote -v [list all added remote repository]

git stash

git log
